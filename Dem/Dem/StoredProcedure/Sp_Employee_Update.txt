CREATE PROC [dbo].[Sp_employee_Update]
@Emp_id int,
@Name nvarchar(max),
@Age int,
@Salary decimal(10, 2)
AS
BEGIN
UPDATE Employees SET Name = @Name, Age=@Age, Salary=@Salary WHERE Id = @Emp_id
END
GO
