CREATE PROC [dbo].[Sp_employee_Add]
@Name nvarchar(max),
@Age int,
@Salary decimal(10, 2),
@Emp_id int out
AS
BEGIN
INSERT INTO Employees(Name, Age, Salary) values(@Name, @Age, @Salary) set @Emp_id = SCOPE_IDENTITY()
return @Emp_id;
END
GO
